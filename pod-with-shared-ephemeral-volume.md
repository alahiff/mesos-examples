Simple example of a pod with two containers with a shared ephemeral volume.

````
{
   "tasks":[
      {
         "name":"test01",
         "task_id":{
            "value":"test01.00"
         },
         "agent_id":{
            "value":""
         },
         "resources":[
            {
               "name":"cpus",
               "type":"SCALAR",
               "scalar":{
                  "value":0.1
               }
            },
            {
               "name":"mem",
               "type":"SCALAR",
               "scalar":{
                  "value":32
               }
            }
         ],
         "command":{
            "value":"touch /opt/sharedpath/test"
         },
         "container":{
            "type":"MESOS",
            "mesos":{
               "image":{
                  "type":"DOCKER",
                  "docker":{
                     "name":"busybox:latest"
                  }
               }
            },
            "volumes":[
               {
                  "container_path":"/opt/sharedpath",
                  "mode":"RW",
                  "source":{
                     "sandbox_path":{
                        "type":"PARENT",
                        "path":"sharedpath",
                        "name":"sharedpath"
                     },
                     "type":"SANDBOX_PATH"
                  }
               }
            ]
         }
      },
      {
         "name":"test02",
         "task_id":{
            "value":"test02.00"
         },
         "agent_id":{
            "value":""
         },
         "resources":[
            {
               "name":"cpus",
               "type":"SCALAR",
               "scalar":{
                  "value":0.5
               }
            },
            {
               "name":"mem",
               "type":"SCALAR",
               "scalar":{
                  "value":1024
               }
            }
         ],
         "command":{
            "value":"sleep 2 ; ls -lt /opt/sharedpath"
         },
         "container":{
            "type":"MESOS",
            "mesos":{
               "image":{
                  "type":"DOCKER",
                  "docker":{
                     "name":"busybox:latest"
                  }
               }
            },
            "volumes":[
               {
                  "container_path":"/opt/sharedpath",
                  "mode":"RW",
                  "source":{
                     "sandbox_path":{
                        "type":"PARENT",
                        "path":"sharedpath",
                        "name":"sharedpath"
                     },
                     "type":"SANDBOX_PATH"
                  }
               }
            ]
         }
      }
   ]
}
````
